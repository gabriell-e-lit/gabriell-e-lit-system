/// AuthorPriorityRules v1

// 1) H2 автори → задължителни биографии в таговете
if author appears as H2 in "Автори в брой X":
    must_have_wp_tag = true
    // ако има биография в тази страница → тя става основна biography
    // wp_tag трябва да съдържа биографията (копиране/синхронизация)
    // first_appearance = тази страница или съответния Issue

// 2) H3 автори → задължителни статични страници
if author appears as H3 in structural pages:
    must_have_static_page = true
    // статичната страница трябва да бъде създадена/свързана
    // static_author_url не трябва да бъде празно

// 3) Причина за H3 (не приоритет, а структура)
// H3 се използва, защото има H2 нива: година, роля, брой, колекция.
// Това НЕ намалява важността на автора, а само описва структурата.

// 4) Приоритет на източниците за biography и portrait
priority_for_biography_and_portrait:
    1. H2 страници "Автори в брой X"
    2. H3 структурни страници (издателство, е-библиотека, е-галерия, гост-редактори)
    3. wp_tag
    4. wp_posts
    5. static_site
    6. ebooks
